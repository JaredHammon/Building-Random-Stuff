        function loadData() {
            const saved = localStorage.getItem('boysNightGaming');
            if (saved) {
                const savedGames = JSON.parse(saved);
                games.forEach((game, i) => {
                    if (savedGames[i]) {
                        game.upvotes = savedGames[i].upvotes || 0;
                        game.downvotes = savedGames[i].downvotes || 0;
                        game.voted = savedGames[i].voted || null;
                        game.reviews = savedGames[i].reviews || [];
                    }
                });
            }
        }

        function saveData() {
            localStorage.setItem('boysNightGaming', JSON.stringify(games));
        }

        function vote(index, type) {
            const game = games[index];
            
            if (game.voted === 'up') game.upvotes--;
            if (game.voted === 'down') game.downvotes--;
            
            if (game.voted === type) {
                game.voted = null;
            } else {
                game.voted = type;
                if (type === 'up') game.upvotes++;
                else game.downvotes++;
            }
            
            saveData();
            render();
        }

        function openModal(index) {
            currentGameIndex = index;
            const game = games[index];
            const modal = document.getElementById('gameModal');
            
            document.getElementById('modalTitle').textContent = game.title;
            document.getElementById('modalIcon').textContent = game.icon;
            document.getElementById('modalPrice').textContent = game.price;
            document.getElementById('modalPlayers').textContent = game.players === 'solo' ? 'Solo' : game.players === 'coop' ? '2-4 Players' : '4+ Players';
            document.getElementById('modalType').textContent = game.type;
            document.getElementById('modalDescription').textContent = game.description;
            document.getElementById('modalSteamLink').href = game.steamLink;
            
            // Render tips
            const tipsList = document.getElementById('modalTips');
            tipsList.innerHTML = '';
            game.tips.forEach(tip => {
                const li = document.createElement('li');
                li.textContent = tip;
                tipsList.appendChild(li);
            });
            
            // Render reviews
            renderReviews(index);
            
            modal.classList.add('active');
        }

        function closeModal() {
            document.getElementById('gameModal').classList.remove('active');
            currentGameIndex = null;
        }

        function renderReviews(index) {
            const game = games[index];
            const container = document.getElementById('reviewsContainer');
            
            if (game.reviews.length === 0) {
                container.innerHTML = '<div class="no-reviews">No reviews yet. Be the first to review!</div>';
                return;
            }
            
            container.innerHTML = '';
            game.reviews.forEach(review => {
                const reviewEl = document.createElement('div');
                reviewEl.className = 'review';
                reviewEl.innerHTML = `
                    <div class="review-author">
                        ${review.author}
                        ${review.author === 'Jake' ? '<span class="review-author-badge">üç∫ HOST</span>' : ''}
                        ${review.author === 'Cole' ? '<span class="review-author-badge">üö¨ HOST</span>' : ''}
                    </div>
                    <div class="review-rating">${'üç∫'.repeat(review.rating)}</div>
                    <div class="review-text">${review.text}</div>
                `;
                container.appendChild(reviewEl);
            });
        }

        function submitReview() {
            const author = document.getElementById('reviewAuthor').value;
            const rating = parseInt(document.getElementById('reviewRating').value);
            const text = document.getElementById('reviewText').value;
            
            if (!author || !text) {
                alert('Please fill in all fields!');
                return;
            }
            
            const game = games[currentGameIndex];
            game.reviews.push({ author, rating, text });
            
            saveData();
            renderReviews(currentGameIndex);
            
            document.getElementById('reviewAuthor').value = '';
            document.getElementById('reviewRating').value = '5';
            document.getElementById('reviewText').value = '';
        }

        function applyFilters() {
            const search = document.getElementById('searchInput').value.toLowerCase();
            const priceFilter = document.getElementById('priceFilter').value;
            const playerFilter = document.getElementById('playerFilter').value;
            const ratingFilter = parseInt(document.getElementById('ratingFilter').value) || 0;
            
            games.forEach((game, index) => {
                let show = true;
                
                // Search filter
                if (search) {
                    show = game.title.toLowerCase().includes(search) || 
                           game.description.toLowerCase().includes(search);
                }
                
                // Price filter
                if (priceFilter) {
                    if (priceFilter === 'free' && game.priceNum !== 0) show = false;
                    if (priceFilter === 'under10' && (game.priceNum === 0 || game.priceNum >= 10)) show = false;
                    if (priceFilter === '10to15' && (game.priceNum < 10 || game.priceNum >= 15)) show = false;
                    if (priceFilter === 'over15' && game.priceNum < 15) show = false;
                }
                
                // Player filter
                if (playerFilter && game.players !== playerFilter) show = false;
                
                // Rating filter
                if (ratingFilter > 0 && game.beerRating && game.beerRating < ratingFilter) show = false;
                
                // Hide the card
                const cards = document.querySelectorAll('[data-index="' + index + '"]');
                cards.forEach(card => {
                    card.style.display = show ? '' : 'none';
                });
            });
        }

        function resetFilters() {
            document.getElementById('searchInput').value = '';
            document.getElementById('priceFilter').value = '';
            document.getElementById('playerFilter').value = '';
            document.getElementById('ratingFilter').value = '';
            applyFilters();
        }

        function createCard(game, index) {
            const card = document.createElement('div');
            card.className = 'game-card' + (game.hallOfFame ? ' hall-of-fame' : '');
            card.setAttribute('data-index', index);
            card.style.cursor = 'pointer';
            card.onclick = () => openModal(index);
            
            let html = '';
            
            html += `<div class="game-icon-container">${game.icon}</div>`;
            
            html += '<div class="game-content">';
            
            if (game.hallOfFame) {
                html += '<span class="legendary-badge">üèÜ LEGENDARY</span>';
            }
            
            html += `<h3 class="game-title">${game.title}</h3>`;
            
            if (game.hallOfFame && game.beerRating) {
                html += `<div class="beer-rating">${'üç∫'.repeat(game.beerRating)}</div>`;
            } else {
                html += `<div class="game-price">${game.price}</div>`;
            }
            
            html += `<p class="game-desc">${game.description}</p>`;
            
            html += `
                <div class="vote-section" onclick="event.stopPropagation();">
                    <button class="vote-btn up ${game.voted === 'up' ? 'active' : ''}" 
                            onclick="vote(${index}, 'up')">üç∫</button>
                    <button class="vote-btn down ${game.voted === 'down' ? 'active' : ''}" 
                            onclick="vote(${index}, 'down')">‚ùå</button>
                </div>
                <div class="vote-count">
                    üç∫ ${game.upvotes} | ‚ùå ${game.downvotes}
                </div>
                <a href="${game.steamLink}" target="_blank" class="steam-link" onclick="event.stopPropagation();">
                    VIEW ON STEAM ‚Üí
                </a>
            `;
            
            html += '</div>';
            card.innerHTML = html;
            return card;
        }

        function render() {
            document.getElementById('hallOfFame').innerHTML = '';
            document.getElementById('freeGames').innerHTML = '';
            document.getElementById('under10').innerHTML = '';
            document.getElementById('range10to15').innerHTML = '';
            document.getElementById('over15').innerHTML = '';
            document.getElementById('comingSoon').innerHTML = '';
            
            games.forEach((game, index) => {
                const card = createCard(game, index);
                
                if (game.hallOfFame) {
                    document.getElementById('hallOfFame').appendChild(card);
                } else if (game.comingSoon) {
                    document.getElementById('comingSoon').appendChild(card);
                } else if (game.priceNum === 0) {
                    document.getElementById('freeGames').appendChild(card);
                } else if (game.priceNum < 10) {
                    document.getElementById('under10').appendChild(card);
                } else if (game.priceNum < 15) {
                    document.getElementById('range10to15').appendChild(card);
                } else if (game.priceNum < 999) {
                    document.getElementById('over15').appendChild(card);
                }
            });
        }

        // Close modal when clicking outside
        window.onclick = function(event) {
            const modal = document.getElementById('gameModal');
            if (event.target == modal) {
                closeModal();
            }
        }

        loadData();
        render();
    </script>
